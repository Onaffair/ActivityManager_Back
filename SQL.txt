create table activity_report
(
    report_id       bigint auto_increment comment '举报ID'
        primary key,
    activity_id     int                                                                                 not null comment '被举报活动ID',
    reporter        varchar(255)                                                                        not null comment '举报人账号',
    report_type     enum ('illegal_content', 'false_information', 'fraud', 'inappropriate', 'other')    not null comment '举报类型',
    description     text                                                                                not null comment '举报描述',
    evidence_images json                                                                                null comment '证据图片(最多5张)',
    report_status   enum ('pending', 'processing', 'resolved', 'rejected') default 'pending'            not null comment '处理状态',
    handler_account varchar(255)                                                                        null comment '处理管理员账号',
    handle_action   enum ('accept', 'reject', 'activity_removed', 'warning_sent', 'user_banned')        null comment '处理措施',
    handle_comment  text                                                                                null comment '处理意见',
    created_at      datetime(3)                                            default CURRENT_TIMESTAMP(3) not null comment '举报时间',
    handled_at      datetime(3)                                                                         null comment '处理时间',
    constraint chk_description_length
        check (char_length(`description`) >= 10),
    constraint chk_evidence_limit
        check (json_length(`evidence_images`) <= 5),
    constraint chk_handle_info
        check (((`report_status` in (_utf8mb4\'resolved\',_utf8mb4\'rejected\')) and (`handler_account` is not null) and (`handle_action` is not null)) or (`report_status` not in (_utf8mb4\'resolved\',_utf8mb4\'rejected\')))
)
comment '活动举报记录表';

create index idx_activity_report
    on activity_report (activity_id, created_at);

create index idx_handler
    on activity_report (handler_account);

create index idx_report_status
    on activity_report (report_status);

create index idx_report_time
    on activity_report (created_at);

create index idx_reporter
    on activity_report (reporter);

create definer = root@localhost trigger after_comment_insert
    after insert
    on activity_comment
    for each row
BEGIN
    -- 更新 activity 表中的 rating
    UPDATE activity
    SET rating = (
        SELECT COALESCE(AVG(rating), 0)
        FROM activity_comment
        WHERE activity_id = NEW.activity_id
    )
    WHERE id = NEW.activity_id;
END;

create table activity_enrollment
(
    enrollment_id   int auto_increment comment '报名ID'
        primary key,
    user_account    varchar(255)                         not null comment '用户账号',
    activity_id     int                                  not null comment '活动ID',
    enrollment_time datetime   default CURRENT_TIMESTAMP null comment '报名时间',
    is_confirmed    tinyint(1) default 1                 null comment '是否确认参加',
    constraint fk_enrollment_activity
        foreign key (activity_id) references activity (id),
    constraint fk_enrollment_user
        foreign key (user_account) references user_table (user_account)
)
    comment '活动报名表';

create index idx_activity_id
    on activity_enrollment (activity_id);

create index idx_user_account
    on activity_enrollment (user_account);

create definer = root@localhost trigger after_enrollment_delete
    after delete
    on activity_enrollment
    for each row
BEGIN
    UPDATE activity
    SET join_num = join_num - 1
    WHERE id = OLD.activity_id;
END;

create definer = root@localhost trigger after_enrollment_insert
    after insert
    on activity_enrollment
    for each row
BEGIN
    UPDATE activity
    SET join_num = join_num + 1
    WHERE id = NEW.activity_id;
END;

create definer = root@localhost trigger after_enrollment_update
    after update
    on activity_enrollment
    for each row
BEGIN
    -- 如果确认状态发生变化
    IF NEW.is_confirmed != OLD.is_confirmed THEN
        -- 更新活动实际参与人数
        UPDATE activity
        SET join_num = join_num + CASE
                                      WHEN NEW.is_confirmed = 1 THEN 1 -- 确认参加
                                      ELSE -1 -- 取消确认
            END,
            updated_at = NOW()
        WHERE id = NEW.activity_id;

        -- 检查人数限制
        UPDATE activity
        SET status = CASE
                         WHEN join_num >= most_join_num THEN 3 -- 报名人数已满
                         WHEN status = 3 AND join_num < most_join_num THEN 2 -- 恢复报名
                         ELSE status
            END
        WHERE id = NEW.activity_id;
    END IF;
END;
